  \tikz[baseline=-0.55ex]\fill[blue!75!black]
    (0,0)--(0.18,0.09)--(0,0.18)--cycle;}

      {\textsc{Content embedding}};

  .. controls +(0,-1.6) and +(1.2,1.2) .. (final.east);
